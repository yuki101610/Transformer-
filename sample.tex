        \footnotesize
        \begin{block}{Input Embedding(埋め込み)}
            総トークン数をVとして、V次元のOne-hotベクトルを埋め込みに変換する行列Eを定義
            \begin{align*}
                E \in \mathbb{R}^{V \times d_{\text{model}}}
            \end{align*}
            t=[0,0..,1,..,0,0]を,第k成分のみ1で、他は全て0のOne-hotベクトルとする
            \begin{align*}
                tE = d
            \end{align*}
            \bm{d}はEのk行目と一致する横ベクトル
        \end{block}
    \end{columns}
    \vspace{1.5em}
    \tiny
    [7]Vaswani, A. et al., Attention Is All You Need, Proceedings of the 31st International Conference on Neural Information Processing Systems, pp.6000 - 6010, 2017.

\begin{frame}{Transformer}
    \framesubtitle{Encoder - Scaled Dot-Product Attention}
    \begin{columns}[T]
        \vspace{-1em}
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Scaled Dot Product Attention.png}
        \end{figure}
        \vspace{-1em}
        \begin{table}[]
            \tiny
            \begin{tabular}{lll}
                \hline
                \textbf{変数} & \textbf{名称} & \textbf{イメージ・定義} \\ \hline
                $Q\in \mathbb{R}^{n \times d_{k}}$ & Query &  「何に注目したいか」 \\
                $K\in \mathbb{R}^{n \times d_{k}}$ & Key & 「どのような情報を持っているか」 \\
                $V\in \mathbb{R}^{n \times d_{v}}$ & Value & 実際に抽出される「情報」 \\
                $n$ & シーケンス長 & 入力トークン数 \\
                $d_k$ & 次元数 & $Q, K$ の次元数 \\
                $d_v$ & 次元数 & $V$ の次元数 \\
                $\text{softmax}$ & 関数 & 正規化する処理 \\ \hline
            \end{tabular}
        \end{table}
        \column{0.5\textwidth}
        \tiny
        \[
        \text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
        \]
        \vspace{0.5em}
        {\tiny \textbf{処理フロー}}
        \begin{itemize}
            \item \textbf{1. MatMul}: $Q$と$K$の内積により単語間の相関（スコア）を算出
            \item \textbf{2. Scale}: $\sqrt{d_k}$で除算。$d_k$増大による勾配消失を防止
            \item \textbf{3. Softmax}: スコアを合計1の確率（重み）に正規化
            \item \textbf{4. MatMul}: 重みに基づき$V$を足し合わせ、文脈を集約
        \end{itemize}
    \end{columns}
    \vspace{2.5em}
    \tiny
    [7]Vaswani, A. et al., Attention Is All You Need, Proceedings of the 31st International Conference on Neural Information Processing Systems, pp.6000 - 6010, 2017.
\end{frame}

\begin{frame}{Transformer}
    \framesubtitle{Encoder - Add \& Norm}
    \begin{columns}[T]
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Encoder.png}
        \end{figure}

        \column{0.65\textwidth}
        \footnotesize
        \begin{block}{Residual Connection \& Layer Norm}
            各サブレイヤの出力を以下の処理を行う
            \begin{align*}
                \text{Output} = \text{LayerNorm}(x + \text{Sublayer}(x))
            \end{align*}
            \begin{itemize}
                \item \textbf{Add (Residual)}: 前の層の出力$\text{Sublayer}(x)$へ入力 $x$ を直接足す
                
                \rightarrow 勾配消失を防ぎ、深い層でも学習が安定
                \item \textbf{LayerNorm}: 特徴ベクトルの平均を0、分散を1に正規化
                
                \rightarrow 学習が高速化・安定化
            \end{itemize}
        \end{block}
    \end{columns}
    \vspace{0.5em}
    \tiny
    [7] Vaswani, A. et al., Attention Is All You Need, Proceedings of the 31st International Conference on Neural Information Processing Systems, pp.6000 - 6010, 2017.
\end{frame}

    \begin{columns}[T]
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Encoder.png}
        \end{figure}

        \column{0.65\textwidth}
        \footnotesize
        \begin{block}{Position-wise Feed Forward (FFN)}
            \begin{align*}
                \text{FFN}(x) = \max(0, xW_1 + b_1)W_2 + b_2
            \end{align*}
            \textbf{変数・次元数定義：}
            \begin{itemize}
                \item $W_1 \in \mathbb{R}^{d_{\text{model}} \times d_{ff}}$ (512 $\to$ 2048)
                \item $W_2 \in \mathbb{R}^{d_{ff} \times d_{\text{model}}}$ (2048 $\to$ 512)
            \end{itemize}
            Attention層から出力された関係性情報を、非線形変換し、より抽象度の高い特徴を抽出する
        \end{block}
    \end{columns}
    \vspace{0.5em}

    [7] Vaswani, A. et al., Attention Is All You Need, Proceedings of the 31st International Conference on Neural Information Processing Systems, pp.6000 - 6010, 2017.

\begin{frame}{Transformer}
    \framesubtitle{Encoder - Positional Encoding}
    \begin{columns}[T]
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Encoder.png}
        \end{figure}

        \column{0.65\textwidth}
        \footnotesize
        \begin{block}{Positional Encoding(位置エンコーディング)}
            単語の順序情報を付与するため、埋め込みベクトルに加算する。
            \begin{align*}
                PE_{(pos, 2i)} &= \sin(pos / 10000^{2i/d_{\text{model}}}) \\
                PE_{(pos, 2i+1)} &= \cos(pos / 10000^{2i/d_{\text{model}}})
            \end{align*}
            \textbf{変数定義：}
            \begin{itemize}
                \item $pos$: 文中でのトークンの位置（0, 1, ...）
                \item $i$: 次元のインデックス（0から $d_{\text{model}}/2$ まで）
                \item $d_{\text{model}}$: モデルの次元数（512）
            \end{itemize}
            \textbf{役割：} RNNのような逐次処理を行わないため、絶対的・相対的な位置関係を正弦波の周期性により表現する。
        \end{block}
    \end{columns}
    \vspace{0.5em}
    \tiny
    [7] Vaswani, A. et al., Attention Is All You Need, NIPS, 2017.
\end{frame}
