            \begin{tabular}{l | c | c}
                \toprule
                & \textbf{先行研究} & \textbf{本研究} \\
                \midrule
                \textbf{入力} & One-hot & \textbf{分散表現} \\
                \textbf{規模} & $N_{\mathrm{rec}} \ge 16,384$ & \textbf{\textcolor{blue}{$N_{\mathrm{rec}}=1,000$}} \\
                \bottomrule
            \end{tabular}%
            }
            
            \vspace{0.5em}
            \raggedright
            \scriptsize
            $\Rightarrow$ 入力に意味情報を持たせることで、リザバー層の負担（記憶・識別）を軽減する。
        \end{block}

        % --- 右：モデルの構成（How） ---
        \column{0.43\textwidth}
        \begin{block}{モデル処理フロー}
            \centering
            \small
            % フローの簡易表現
            \textbf{単語} $w_t$ \\
            $\downarrow$ \scriptsize{\textit{Word2Vec (事前学習)}}\\
            \small \textbf{入力} $\bm{d}(t)$ \\
            $\downarrow$ \scriptsize{\textit{Input・Reservoir (固定)}}\\
            \small \textbf{状態} $\bm{r}(t)$ \\
            $\downarrow$ \scriptsize{\textit{Readout (学習)}}\\
            \small \textbf{予測} $\bm{y}(t) \approx \bm{d}(t+1)$
        \end{block}

        \vspace{0.5em}
        \footnotesize
        \structure{\textbf{構成の要点}}
        \scriptsize
        \begin{itemize}
            \item \textbf{入力層}: 事前学習済みの\\分散表現を使用し、意味情報を入力(重み固定)
            \item \textbf{リザバー層}: 入力を高次元\\非線形空間へ写像（重み固定）
            \item \textbf{出力層}: 次の単語の分散表現を直接予測するように出力層の重みを学習
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{手法 - 入力表現の高度化}
    \vspace{1em}
    %\framesubtitle{Word2Vec (Skip-gram) による意味情報の導入}
    
    入力層の次元削減および意味情報の付与を目的に、Word2Vec[5]のSkip-gramアルゴリズムを採用する

    \vspace{1em}
    \footnotesize
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Skip-gramの役割}
            中心語から文脈語を予測する学習を通じ、\textbf{単語間の意味的類似性}をベクトル空間上の距離として取得
            \vspace{0.5em}
            \begin{center}
                \footnotesize
                「王」$-$「男」$+$「女」$\approx$「女王」
            \end{center}
        \end{block}

        \column{0.48\textwidth}
        \structure{\textbf{本研究における設定}}
        \begin{itemize}
            \item \textbf{パラメータ固定}:\\
            RC学習中は固定し、1度の実行に書かる計算コストを抑制
            \item \textbf{次元数 ($d_{\mathrm{emb}}$) の検証}:\\
            \textbf{32, 64, 128} の3パターンで比較し、表現能力と計算負荷のトレードオフを明らかにする
        \end{itemize}
    \end{columns}
    \vspace{3em}
    \tiny
    [5]Mikolov, T. et al., Efficient Estimation of Word Representations in Vector Space, Proceedings of ICLR, 2013.
\end{frame}

\begin{frame}{手法 - リザバー計算の定式化}
    %\framesubtitle{状態更新と学習則}
    \vspace{1em}
    \begin{columns}[t]
        % --- 左カラム：数式 ---

        \column{.55\textwidth}
        \footnotesize
        \structure{1. 状態更新}
        \begin{flalign*}
            \bm{x}(t) &= (1 - \alpha)\bm{x}(t - 1) & \\
                    &\quad + \alpha \Big( {W}^{\mathrm{in}} \bm{d}(t) + {W}^{\mathrm{rec}} \bm{r}(t - 1) \Big) &
        \end{flalign*}

        \vspace{-1.5em}
        \begin{flalign*}
            \bm{r}(t) &= \tanh(\bm{x}(t)) &
        \end{flalign*}
        
        \vspace{1.5em}
        \structure{2. 出力}
        \begin{flalign*}
            \bm{y}(t) &= W^{\mathrm{out}} \bm{r}(t) &
        \end{flalign*}
        
        \vspace{1.5em}
        \structure{3. 重み更新}
        \begin{flalign*}
            W^{\mathrm{out}}(t+1) &= W^{\mathrm{out}}(t) & \\
                                &\quad + \frac{\mu}{B} \sum_{k=1}^{B} \bm{e}_k(t) \bm{r}_k(t)^\top &
        \end{flalign*}
        % --- 右カラム：変数定義 ---
        \column{.4\textwidth}
        \scriptsize
        \begin{block}{変数定義}
            \begin{itemize}
                \item $\bm{d}(t)$: 入力ベクトル
                \item $\bm{r}(t)$: リザバー状態ベクトル
                \item $\bm{y}(t)$: 予測出力ベクトル
                \item $\bm{e}_k(t)$: 予測誤差 ($\bm{y}_{\mathrm{true}} - \bm{y}$)
                \item $\alpha$: 漏れ率
                \item $\mu$: 学習率
                \item $B$: ミニバッチサイズ
            \end{itemize}
        \end{block}
        
        \vspace{0.1em}
        \begin{alertblock}{学習対象}
            $W^{\mathrm{out}}$ のみ更新し、\\$W^{\mathrm{in}}, W^{\mathrm{rec}}$ はランダムかつスパースに固定する
        \end{alertblock}
    \end{columns}
\end{frame}

\begin{frame}{手法 - 学習の効率化}
    %\framesubtitle{語彙数 $V$ に依存しない出力層の設計}

    %次単語の「カテゴリ分類」ではなく、\textbf{「分散表現の回帰」}を採用することで、出力層の計算量およびメモリ消費量を抜本的に低減する。

    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{alertblock}{従来手法}
            %\footnotesize
            \scriptsize
            \begin{itemize}
                \item ターゲット：one-hotベクトル
                \item 語彙数 $V$ に比例した出力行列が必要
                \item 学習・推論共に $V$ 次元の演算を伴う
                \item \textbf{計算量：$O(V)$} \hfill \textcolor{red}{\textbf{（高負荷）}}
            \end{itemize}
        \end{alertblock}

        \column{0.48\textwidth}
        \begin{exampleblock}{提案手法}
            %\footnotesize
            \scriptsize
            \begin{itemize}
                \item ターゲット：分散表現ベクトル
                \item 埋め込み次元 $d$ にのみ依存
                \item $d \ll V$
                \item \textbf{計算量：$O(d)$} \hfill \textcolor{blue}{\textbf{（低負荷）}}
            \end{itemize}
        \end{exampleblock}
    \end{columns}

    \vspace{1.5em}
    \centering
    \large
    $\Downarrow$
    \vspace{0.5em}
    
    \small
    \textbf{ミニバッチ学習を用いた逐次的な重み更新}により、\\
    大規模なデータセットに対しても高速かつ省メモリな学習を実現
\end{frame}

\begin{frame}{手法 - 評価手法 - 次単語予測精度 (Perplexity)}
    %\framesubtitle{分散表現の類似度に基づく確率分布の生成}
    \small
    予測ベクトル $\hat{\bm{y}}(t)$ と各語彙が持つ分散表現 $\bm{v}$ とのコサイン類似度に基づき確率分布を算出する
    \vspace{0.5em}
    \begin{columns}[t]
        % --- 左：数式定義 ---
        \column{0.55\textwidth}
        \footnotesize
        \structure{1. 類似度による確率定義}
        \begin{align*}
            p(w_t | \bm{r}(t)) &= \frac{\exp \left( \frac{\operatorname{cos\_sim}(\hat{\bm{y}}(t), \bm{v}_{t})}{\tau} \right)}{\sum_{k=1}^{|V|} \exp \left( \frac{\operatorname{cos\_sim}(\hat{\bm{y}}(t), \bm{v}_{k})}{\tau} \right)}
        \end{align*}
        
        \structure{2. Perplexity (PPL) の算出}
        \begin{align*}
            \mathrm{PPL}(W) &= \exp \left( - \frac{1}{N} \sum_{t=1}^{N} \log p(w_t | \bm{r}(t)) \right)
        \end{align*}

        % --- 右：変数定義とプロセス ---
        \column{0.42\textwidth}
        \vspace{-1em}
        \scriptsize
        \begin{block}{変数定義}
            \begin{itemize}
                \item $\hat{\bm{y}}(t)$: 時刻 $t$ における\textbf{予測ベクトル}
                \item $\bm{v}_{t}$: 正解単語 $w_t$ の\textbf{分散表現}
                \item $\bm{v}_{k}$: 語彙集合 $V$ 内の $k$ 番目の\textbf{分散表現}
                \item $\tau$: 温度パラメータ
                \item $N$: テストデータの総単語数-10
            \end{itemize}
        \end{block}
    \end{columns}
    \vspace{1em}
    \hrule
    \vspace{0.5em}
    \scriptsize
    \structure{\textbf{評価プロセス}}
        \begin{itemize}
            \item \textbf{文脈長}: 10単語
            \item \textbf{評価対象}:
            $t=1 \sim 10$ は内部状態の更新のみを行い、$t=11$ 以降の単語をPPL計算の対象とする。
        \end{itemize}
\end{frame}

\begin{frame}{手法 - 評価手法 - 文法性判断 (BLiMP)[5]}
    %\framesubtitle{Minimal Pairを用いた言語現象の習得度評価}
    \vspace{.5em}
    文法的に正当な文 ($W_{\mathrm{good}}$) と誤った文 ($W_{\mathrm{bad}}$) のペアに対し、モデルの\textbf{文スコア}を比較して判定を行う

    %\vspace{1em}

    \begin{columns}[T]
        % --- 左：スコア定義 ---
        \column{0.55\textwidth}
        \small
        \begin{block}{文スコア}
            文全体を通しての対数尤度の総和
            \vspace{-0.5em}
            \begin{align*}
                \operatorname{Score}(W) = \sum_{t=1}^{T-1} \log p(w_{t+1} \mid \bm{r}(t))
            \end{align*}
            \vspace{-0.5em}
            \scriptsize
            \begin{itemize}
                \item \textbf{$T$}: 評価対象の文脈長 (文の単語数)
                \item $p(\cdot)$: PPLと同様に類似度に基づき算出
            \end{itemize}
        \end{block}

        % --- 右：判定基準 ---
        \column{0.41\textwidth}
        \small
        \begin{exampleblock}{判定基準}
            正しい文法知識があれば、正当な文の尤度が高くなる
            
            \vspace{0.5em}
            \centering
            \textbf{正解条件:}
            \begin{align*}
                \operatorname{Score}(W_{\mathrm{good}}) > \operatorname{Score}(W_{\mathrm{bad}})
            \end{align*}
        \end{exampleblock}
    \end{columns}

    \vspace{1em}
    \hrule
    \vspace{0.5em}

    \scriptsize
    \textbf{評価対象データ}:\\
    英語の\textbf{形態論・統語論・意味論}に基づく12種類の言語現象（全67データセット）\\
    各データセットは1,000組のミニマルペアから構成され、現象ごとの正解率を集計する
    \vspace{3em}
    \tiny
    [5]Warstadt, A. et al., BLiMP: The Benchmark of Linguistic Minimal Pairs for English, MIT Press, 2020.
\end{frame}

\begin{frame}{まとめ - 提案モデルの特性と限界}
    \begin{columns}[T]

        \column{0.32\textwidth}
        \begin{block}{1. 予測性能(PPL)の考察}
            \scriptsize
            \begin{itemize}
                \item 「過度な詳細化($D_{\mathrm{emb}}=128$)」は\\学習の妨げとなる\\可能性が示唆される
                \item $\rho \uparrow$ に伴うPPL悪化は、遠い過去情報が残り、単語予測における「干渉」として作用した可能性が\\示唆される
            \end{itemize}
        \end{block}

        % --- 中央：BLiMP全体傾向 ---
        \column{0.32\textwidth}
        \begin{alertblock}{2. 言語性能(BLiMP)全体像}
            \scriptsize
            \begin{itemize}
                \item \textbf{スコアの停滞}: 
                全$\rho$において平均正解率は55\%前後で飽和
                \item $\rho$ の変化に対する平均値の変動が極めて小さいのは、タスク間の精度変動が互いに打ち消し合うため
            \end{itemize}
        \end{alertblock}

        % --- 右：BLiMPタスク別・限界 ---
        \column{0.32\textwidth}
        \begin{exampleblock}{3. 言語現象別の相反関係}
            \scriptsize
            \begin{itemize}
                \item $\rho\uparrow$に伴い精度が向上する長期記憶優位のタスクと精度が低下する短期記憶優位のタスクが確認された。また、評価指標によって適切な次元数も異なる。
                \item 単一の $\rho$ で「短期記憶」と「長期記憶」を両立させることは困難
            \end{itemize}
        \end{exampleblock}
    \end{columns}

    \vspace{1em}
    \centering
    \scriptsize
    \textbf{結論}: リザバーの物理的特性（$\rho$）は言語現象ごとにトレードオフを引き起こす。\\
    単一設定での全タスク網羅は困難であり、複数時間スケールの統合が必要
\end{frame}

    \end{columns}

\begin{frame}{Transformer}
    \framesubtitle{Encoder - Scaled Dot-Product Attention}
    \begin{columns}[T]
        \vspace{-1em}
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Scaled Dot Product Attention.png}
        \end{figure}
        \vspace{-1em}
        \begin{table}[]
            \tiny
            \begin{tabular}{lll}
                \hline
                \textbf{変数} & \textbf{名称} & \textbf{イメージ・定義} \\ \hline
                $Q\in \mathbb{R}^{n \times d_{k}}$ & Query &  「何に注目したいか」 \\
                $K\in \mathbb{R}^{n \times d_{k}}$ & Key & 「どのような情報を持っているか」 \\
                $V\in \mathbb{R}^{n \times d_{v}}$ & Value & 実際に抽出される「情報」 \\
                $n$ & シーケンス長 & 入力トークン数 \\
                $d_k$ & 次元数 & $Q, K$ の次元数 \\
                $d_v$ & 次元数 & $V$ の次元数 \\
                $\text{softmax}$ & 関数 & 正規化する処理 \\ \hline
            \end{tabular}
        \end{table}
        \column{0.5\textwidth}
        \tiny
        \[
        \text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
        \]
        \vspace{0.5em}
        {\tiny \textbf{処理フロー}}
        \begin{itemize}
            \item \textbf{1. MatMul}: $Q$と$K$の内積により単語間の相関（スコア）を算出
            \item \textbf{2. Scale}: $\sqrt{d_k}$で除算。$d_k$増大による勾配消失を防止
            \item \textbf{3. Softmax}: スコアを合計1の確率（重み）に正規化
            \item \textbf{4. MatMul}: 重みに基づき$V$を足し合わせ、文脈を集約
        \end{itemize}
    \end{columns}
    \vspace{2.5em}
    \tiny
    [7]Vaswani, A. et al., Attention Is All You Need, Proceedings of the 31st International Conference on Neural Information Processing Systems, pp.6000 - 6010, 2017.
\end{frame}

        \begin{block}{Residual Connection \& Layer Norm}
            各サブレイヤの出力を以下の処理を行う
            \begin{align*}
                \text{Output} = \text{LayerNorm}(x + \text{Sublayer}(x))
            \end{align*}
            \begin{itemize}
                \item \textbf{Add (Residual)}: 前の層の出力$\text{Sublayer}(x)$へ入力 $x$ を直接足す
                
                \rightarrow 勾配消失を防ぎ、深い層でも学習が安定
                \item \textbf{LayerNorm}: 特徴ベクトルの平均を0、分散を1に正規化
                
                \rightarrow 学習が高速化・安定化
            \end{itemize}
        \end{block}

    \begin{columns}[T]

        \end{figure}
        \column{0.45\textwidth}
        アテンション機構とは
        \begin{itemize}
            \item リザバー状態 $\bm{r}(t)$ に基づき、出力重みを\textbf{動的に計算}する仕組み
            \item 毎時刻で「リザバーのどのノードを優先すべきか」を適応的に決定
        \end{itemize}
    \end{columns}

\end{frame}

\begin{frame}{手法 - アテンション強化型リザーバの定式化}
    \vspace{1em}
    \begin{columns}[t]
        % --- 左カラム：数式 ---
        \column{.6\textwidth}
        \footnotesize
        \structure{1. アテンション重みの動的生成}
        \begin{flalign*}
            \bm{h}(t) &= \text{ReLU}(W_1 \bm{r}(t) + \bm{b}_1) & \\
            W_{\text{att}}(t) &= W_2 \bm{h}(t) + \bm{b}_2 &
        \end{flalign*}
        \textit{\scriptsize ※ $W_{\text{att}}(t)$ は時刻 $t$ ごとに計算される行列}

        \vspace{1.0em}
        \structure{2. 出力}
        \begin{flalign*}
            \bm{y}(t) &= W_{\text{att}}(t) \bm{r}(t) &
        \end{flalign*}

        \vspace{1.0em}
        \structure{3. パラメータ更新 (勾配降下法)}
        \begin{flalign*}
            W_{\text{net}} &\leftarrow W_{\text{net}} - \gamma \nabla \mathcal{L} & 
        \end{flalign*}
        \vspace{-1.5em}
        \begin{itemize}
            \item[] $\mathcal{L} = \text{NRMSE}(\bm{y}, \bm{y}_{\text{true}})$
        \end{itemize}

        % --- 右カラム：変数定義 ---
        \column{.38\textwidth}
        \scriptsize
        \begin{block}{追加・変更変数定義}
            \begin{itemize}
                \item $W_{\text{net}}$: 学習対象$\{W_1, W_2, \bm{b}_1, \bm{b}_2\}$ の総称
                \item $W_{\text{att}}(t)$: アテンション重み
                \item $\bm{h}(t)$: 隠れ層の特徴ベクトル
                \item $\gamma$: 学習率
                \item $\mathcal{L}$: 損失関数 (正規化RMSE)
            \end{itemize}
        \end{block}
    \end{columns}
\end{frame}

\begin{frame}{Transformer}
    \framesubtitle{Encoder - Positional Encoding}
    \begin{columns}[T]
        \column{0.3\textwidth}
        \begin{figure}[bthp]
        \centering
        \includegraphics[width=0.9\linewidth]{Encoder.png}
        \end{figure}

        \column{0.65\textwidth}
        \footnotesize
        \begin{block}{Positional Encoding(位置エンコーディング)}
            単語の順序情報を付与するため、埋め込みベクトルに加算する。
            \begin{align*}
                PE_{(pos, 2i)} &= \sin(pos / 10000^{2i/d_{\text{model}}}) \\
                PE_{(pos, 2i+1)} &= \cos(pos / 10000^{2i/d_{\text{model}}})
            \end{align*}
            \textbf{変数定義：}
            \begin{itemize}
                \item $pos$: 文中でのトークンの位置（0, 1, ...）
                \item $i$: 次元のインデックス（0から $d_{\text{model}}/2$ まで）
                \item $d_{\text{model}}$: モデルの次元数（512）
            \end{itemize}
            \textbf{役割：} RNNのような逐次処理を行わないため、絶対的・相対的な位置関係を正弦波の周期性により表現する。
        \end{block}
    \end{columns}

\end{frame}
